399  std::string ColorText(const std::string& text, Color color) {
400  switch (color) {
401  case Color::red:
402  return std::string("\033[31m") + text + std::string("\033[0m");
403  case Color::green:
404  return std::string("\033[32m") + text + std::string("\033[0m");
405  case Color::yellow:
406  return std::string("\033[33m") + text + std::string("\033[0m");
407  case Color::blue:
408  return std::string("\033[34m") + text + std::string("\033[0m");
409  case Color::magenta:
410  return std::string("\033[35m") + text + std::string("\033[0m");
411  case Color::cyan:
412  return std::string("\033[36m") + text + std::string("\033[0m");
413  case Color::white:
414  return std::string("\033[37m") + text + std::string("\033[0m");
415  case Color::shell_default:
416  return std::string("\033[;0m") + text + std::string("\033[0m");
417  return text;
418  default:
419  return text;
420  }
421  }
