39 /*
40  * Lookup Class with static binomial coeffient
41  *
42  * Create a Lookup Table at compile time and provides fast and efficient look up
43  * of binomial coefficient. Maximum degree directly correlates with the maximum
44  * degree of the bezier spline to be considered.
45  */
46 class FastBinomialCoefficient {
47  private:
49  using ScalarType = std::size_t;
50 
52  constexpr static auto n_values =
53  BinomialCoefficientLookupTableCreator<ScalarType,
54  MAX_BINOMIAL_DEGREE>::n_values;
55 
57  constexpr static std::array<ScalarType, n_values> look_up =
58  BinomialCoefficientLookupTableCreator<
59  ScalarType, MAX_BINOMIAL_DEGREE>::calculate_table();
60 
61  public:
63  constexpr static ScalarType choose(const unsigned int n,
64  const unsigned int i) {
65  assert(n < MAX_BINOMIAL_DEGREE);
66  return look_up[n * (n + 1) / 2 + i];
67  }
68 };
