106  bezier_spline)
107  : weighted_spline_{bezier_spline} {
108  // Initialize new weight function spline
109  weight_function_ =
110  BezierSpline<parametric_dimension, ScalarType, ScalarType>{
111  // Degrees can be taken from weighted spline
112  weighted_spline_.GetDegrees(),
113  // Generate control weights (just 1 for every point)

140  constexpr RationalBezierSpline(
141  const std::array<std::size_t, parametric_dimension> deg,
142  const std::vector<PhysicalPointType_> control_point_vector,
143  const std::vector<ScalarType> weights)
144  : weighted_spline_{deg}, weight_function_{deg, weights} {
145  assert(control_point_vector.size() == weights.size());
146  assert(weighted_spline_.GetNumberOfControlPoints() == weights.size());
147  for (std::size_t i_point{0};
148  i_point < weighted_spline_.control_points.size(); i_point++) {
149  weighted_spline_.control_points[i_point] =
150  control_point_vector[i_point] * weights[i_point];
151  }
152  assert(CheckSplineCompatibility());
153  }
