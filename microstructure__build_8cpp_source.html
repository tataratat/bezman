43 PolynomialBeziers SimpleCrossTile(const double thickness) {

79 RationalBeziers CircleGroup(const double innerR, const double outerR,
80  const int segments, const double arc_degrees) {
82  RationalBeziers ringsegments(segments);
83  const double PI = acos(-1);
84  const double degrees_per_segment = arc_degrees / segments;
85  const double single_weight{std::sin(PI / 2. - degrees_per_segment / 2.)};
86  const double outdor_fact = 1. / single_weight;
87  const std::array<std::size_t, 2> degrees{2, 1};
88  std::vector<double> weights{1, single_weight, 1, 1, single_weight, 1};
89  for (int i_segment{}; i_segment < segments; i_segment++) {
90  const double startsin = std::sin(degrees_per_segment * i_segment);
91  const double startcos = std::cos(degrees_per_segment * i_segment);
92  const double middlesin = std::sin(degrees_per_segment * (i_segment + .5));
93  const double middlecos = std::cos(degrees_per_segment * (i_segment + .5));
94  const double endsin = std::sin(degrees_per_segment * (i_segment + 1.));
95  const double endcos = std::cos(degrees_per_segment * (i_segment + 1.));
96  std::vector<Point2D> ctps{Point2D{startcos * outerR, startsin * outerR},
97  Point2D{middlecos * outdor_fact * outerR,
98  middlesin * outdor_fact * outerR},
99  Point2D{endcos * outerR, endsin * outerR},
100  Point2D{startcos * innerR, startsin * innerR},
101  Point2D{middlecos * outdor_fact * innerR,
102  middlesin * outdor_fact * innerR},
103  Point2D{endcos * innerR, endsin * innerR}};
104 
105  ringsegments[i_segment] = RationalBezier{degrees, ctps, weights};
106  }
107  return ringsegments;
108 }
