36 /*
37  * Point class
38  *
39  * Class inherits most of its functionality from std::array, however, scalar
40  * multiplication and basic operations were added to facilitate the
41  * interpolation of control points in the splines class.
42  *
43  * Note: Class is minimalistic and extended as required
44  *
45  * @TODO : Write unit tests for this class
46  *
47  * @tparam spatial_dimension Spatial dimension of the point
48  * @tparam BaseType Scalar Type describing the coordinates (to impose
49  * CD-Types easier)
50  */
51 template <std::size_t spatial_dimension, typename BaseType = double>
52 class Point : public std::array<BaseType, spatial_dimension> {
53  public:
55  constexpr static unsigned int kSpatialDimension = spatial_dimension;
56 
58  using ScalarType_ = BaseType;
59 
61  constexpr Point() = default;
62 
64  template <typename... scalars>
65  explicit constexpr Point(const scalars&... coords)
66  : std::array<BaseType, spatial_dimension>{coords...} {
67  static_assert(sizeof...(coords) == spatial_dimension,
68  "You are instantiating a Point object with "
69  "more or less than spatial_dimension components.");
70  }
71 
73  constexpr Point operator+(const Point& rhs) const {
74  Point<spatial_dimension, BaseType> sum{(*this)};
75  sum += rhs;
76  return sum;
77  }
78 
80  constexpr Point& operator+=(const Point& rhs) {
81  for (unsigned int i = 0; i < spatial_dimension; i++) {
82  (*this)[i] = (*this)[i] + rhs[i];
83  }
84  return (*this);
85  }
86 
88  constexpr Point operator-(const Point& rhs) const {
89  Point<spatial_dimension, BaseType> difference{(*this)};
90  difference -= rhs;
91  return difference;
92  }
93 
95  constexpr Point& operator-=(const Point& rhs) {
96  for (unsigned int i = 0; i < spatial_dimension; i++) {
97  (*this)[i] = (*this)[i] - rhs[i];
98  }
99  return (*this);
100  }
101 
103  constexpr Point operator-() const {
104  Point inverted_point{(*this)};
105  return inverted_point * static_cast<BaseType>(-1);
106  }

109  constexpr Point& operator*=(const BaseType& scale) {
110  for (unsigned int i = 0; i < spatial_dimension; ++i) (*this)[i] *= scale;
111  return (*this);
112  }
113 
115  constexpr Point operator*(const BaseType& scale) const {
116  Point<spatial_dimension, BaseType> product{(*this)};
117  product *= scale;

143  for (unsigned int i = 0; i < spatial_dimension; ++i)
144  division[i] = scale / point[i];
145  return division;

149  template <typename Scalar>
150  decltype(Scalar{} * BaseType{}) operator*(
151  const Point<spatial_dimension, Scalar>& point) const {
152  using ScalarReturn = decltype(Scalar{} * BaseType{});
153  ScalarReturn result{};
154  for (unsigned int i{}; i < spatial_dimension; i++) {
155  result += (*this)[i] * point[i];
156  }
157  return result;
158  }

161  BaseType SquaredEuclidianNorm() const {
