86  template <
87  typename BaseType,
88  std::enable_if_t<!type_traits::isPoint_v<BaseType>, void*> = nullptr>
89  static std::string Encode(const std::vector<BaseType>& data_vector) {
90  const ByteRepresentation* vector_as_bytes =
91  reinterpret_cast<const ByteRepresentation*>(&data_vector[0]);
92 
93  // Number of bytes for an entry
94  constexpr const std::size_t length_of_entry{sizeof(BaseType{})};
95  // Minimum number of bytes required
96  const std::size_t minimum_n_bytes_required =
97  length_of_entry * data_vector.size();
98  // Number of bytes must be divisible by three
99  const std::size_t additional_padding_bytes =
100  (3 - minimum_n_bytes_required % 3) % 3;
101  // Required groups of three
102  const std::size_t number_of_groups =
103  (minimum_n_bytes_required + additional_padding_bytes) / 3;
104 
105  // Initialize return value
106  std::string encoded_string;
107  encoded_string.resize(number_of_groups * 4);
108 
109  // Loop over bytes and decode them
110  for (std::size_t i_group{}; i_group < number_of_groups; i_group++) {
111  const std::size_t buffer_index = i_group * 3;
112  std::array<ByteRepresentation, 3> buffer{};
113  buffer[0] = buffer_index < minimum_n_bytes_required
114  ? vector_as_bytes[buffer_index + 0]
115  : 0;
116  buffer[1] = buffer_index < minimum_n_bytes_required
117  ? vector_as_bytes[buffer_index + 1]
118  : 0;
119  buffer[2] = buffer_index < minimum_n_bytes_required
120  ? vector_as_bytes[buffer_index + 2]
121  : 0;
122 
123  encoded_string[i_group * 4 + 0] =
124  char_encode_table[((buffer[0] & 0xfc) >> 2)];
125  encoded_string[i_group * 4 + 1] =
126  char_encode_table[((buffer[0] & 0x03) << 4) +
127  ((buffer[1] & 0xf0) >> 4)];
128  encoded_string[i_group * 4 + 2] =
129  char_encode_table[((buffer[1] & 0x0f) << 2) +
130  ((buffer[2] & 0xc0) >> 6)];
131  encoded_string[i_group * 4 + 3] =
132  char_encode_table[((buffer[2] & 0x3f) << 0)];
133  }
134 
135  // Replace trailing invalid data with =
136  for (size_t i = 0; i < additional_padding_bytes; ++i) {
137  encoded_string[number_of_groups * 4 - i - 1] = '=';
138  }
139 
140  return encoded_string;
141  }

197  template <
198  typename OutputType,
199  std::enable_if_t<!type_traits::isPoint_v<OutputType>, void*> = nullptr>
200  static std::vector<OutputType> Decode(const std::string& base64string) {
201  // Check validity of string
202  assert(base64string.size() % 4 == 0);
203 
204  // Init return value
205  const std::size_t number_of_groups{base64string.size() / 4};
206  constexpr const std::size_t length_of_entry{sizeof(OutputType{})};
207  const std::size_t number_of_output_values{(number_of_groups * 3) /
208  length_of_entry};
209  std::vector<OutputType> return_value;
210  return_value.resize(number_of_output_values);
211 
212  // Access as byte stream
213  ByteRepresentation* vector_as_bytes =
214  reinterpret_cast<ByteRepresentation*>(&return_value[0]);
215 
216  // Start the reverse process
217  for (std::size_t i_group{}; i_group < number_of_groups; i_group++) {
218  const std::size_t buffer_index = i_group * 4;
219  std::array<unsigned, 4> buffer{};
220  for (unsigned i{}; i < 4; i++) {
221  buffer[i] = base64string[buffer_index + i] != '='
222  ? char_decode_table[static_cast<unsigned>(
223  base64string[buffer_index + i])]
224  : 255;
225  }
226 
227  // Write bytes
228  if (buffer[1] != 255) {
229  vector_as_bytes[i_group * 3] =
230  ((buffer[0] & 0x3f) << 2) + ((buffer[1] & 0x30) >> 4);
231  }
232  if (buffer[2] != 255) {
233  vector_as_bytes[i_group * 3 + 1] =
234  ((buffer[1] & 0x0f) << 4) + ((buffer[2] & 0x3c) >> 2);
235  }
236  if (buffer[3] != 255) {
237  vector_as_bytes[i_group * 3 + 2] =
238  ((buffer[2] & 0x03) << 6) + ((buffer[3] & 0x3f) >> 0);
239  }
240  }
241  return return_value;
242  }
